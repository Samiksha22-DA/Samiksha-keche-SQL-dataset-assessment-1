create table employee_Information(employee_id int, first_name varchar,	last_name varchar,	
salary int,	hire_date date,	manager_id int,	city varchar,	state varchar,	country varchar
)

create table Sales_Records(sale_id int,	product_id int,	customer_id int,
sale_date date,	quantityn int,	unit_price real,	discount real,	total_sale real,	salesperson_id int,	region varchar
)


create table Product_Details(product_id int,	product_name varchar,	category varchar,	price real,	stock_quantity int,
supplier_id int,	reorder_level int,	discontinued varchar,	created_at date,	updated_at date
)

create table Customer_Information(customer_id int,	first_name varchar,	last_name varchar,	
email varchar,	phone varchar, address varchar,	city varchar,	state varchar,	country varchar,	account_creation_date date
)

create table Order_Details(order_id int, customer_id int,	order_date date,	shipping_date date,
status varchar,	total_amount real,	payment_method varchar,	billing_address varchar,	shipping_address varchar,	shipper_id int
)
